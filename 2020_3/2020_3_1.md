#### [225. 用队列实现栈](https://leetcode-cn.com/problems/implement-stack-using-queues/)

1.两个队列实现栈
//栈是后进先出,也就是说最后入栈的元素需要第一个出战
//队列是后进后出,也就是说出栈的操作实际上是要想办法弹出队列最后添加的元素
//que1作为出栈所需队列,que2作为辅助保存que1队列的不出栈的元素
class MyStack {
public:
    queue<int> que1;
    queue<int> que2;
    /** Initialize your data structure here. */
    MyStack() {

    }
    
    /** Push element x onto stack. */
    void push(int x) {
        que1.push(x);
    }
    
    /** Removes the element on top of the stack and returns that element. */
    int pop() {

        while (que1.size() > 1) {     //出栈que1最后一个元素
            que2.push(que1.front());  //不出栈的元素交给que2暂时保存
            que1.pop();
        }
        int res = que1.front(); //得到出栈元素
        que1 = que2;			//que2返还que1的剩余元素
        while (!que2.empty()) que2.pop();	//清空que2,以便下次使用
        return res;
    }
    
    /** Get the top element. */
    int top() {
        // int res = this->pop();
        // que1.push(res);
        // return res; //这个也可以
        return que1.back();	//因为除了pop()时,元素全部都是在que1中,que2为空的状态
                            //que1尾部元素也就是栈顶元素
    }
    
    /** Returns whether the stack is empty. */
    bool empty() {
        return que1.empty();
    }
};

 2.一个队列实现栈
 class MyStack {
public:
    queue<int> que1;
    /** Initialize your data structure here. */
    MyStack() {

    }
    
    /** Push element x onto stack. */
    void push(int x) {
        que1.push(x);
    }
    
    /** Removes the element on top of the stack and returns that element. */
    int pop() {

        int n = que1.size();
        for (int i = 1; i < n; ++i) {
            que1.push(que1.front());
            que1.pop();
        }
        int res = que1.front();
        que1.pop();
        return res;
    }
    
    /** Get the top element. */
    int top() {
        return que1.back();
    }
    
    /** Returns whether the stack is empty. */
    bool empty() {
        return que1.empty();
    }
};

        deq.pop_back();
