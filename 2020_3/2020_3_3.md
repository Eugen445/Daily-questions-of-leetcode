#### [面试题 10.01. 合并排序的数组](https://leetcode-cn.com/problems/sorted-merge-lcci/)

```
1.sort排序 O(nlogn) O(1)
class Solution {
public:
    void merge(vector<int>& A, int m, vector<int>& B, int n) {

        for (int i = 0; i < n; ++i)
            A[m + i] = B[i];
        sort(A.begin(), A.end());
    }
};

2.正向双指针 O(n) O(n)
class Solution {
public:
    void merge(vector<int>& A, int m, vector<int>& B, int n) {

        int APos = 0, BPos = 0;
        int C[m + n];
        int pos = 0;
        while (APos < m || BPos < n) {
            if (APos >= m) C[pos++] = B[BPos++];
            else if (BPos >= n) C[pos++] = A[APos++];
            else if (A[APos] < B[BPos]) C[pos++] = A[APos++];
            else C[pos++] = B[BPos++];
        }

        for (int i = 0; i < m + n; ++i)
            A[i] = C[i];
    }
};

3.逆向双指针 O(n) O(1)
class Solution {
public:
    void merge(vector<int>& A, int m, vector<int>& B, int n) {

        int APos = m - 1, BPos = n - 1;
        int pos = m + n - 1;
        while (APos >= 0 || BPos >= 0) {
            if (APos < 0) A[pos--] = B[BPos--];
            else if (BPos < 0) A[pos--] = A[APos--];
            else if (A[APos] < B[BPos]) A[pos--] = B[BPos--];
            else A[pos--] = A[APos--]; 
        }
    }
};
```
